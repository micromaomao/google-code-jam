func start() {
	var t int
	mustReadLineOfInts(&t)
	for i := 0; i < t; i++ {
		var N int
		mustReadLineOfInts(&N)
		lPath := mustReadLine()
		stdout.WriteString("Case #")
		stdout.WriteString(strconv.Itoa(i + 1))
		stdout.WriteString(": ")
		for i := 0; i < len(lPath); i++ {
			if lPath[i] == 'E' {
				stdout.WriteByte('S')
			} else {
				stdout.WriteByte('E')
			}
		}
		stdout.WriteByte('\n')
	}
}
