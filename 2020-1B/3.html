fn case(i: usize, mut r: usize, s: usize) {
    let mut res: Vec<(usize, usize)> = Vec::new();

    while r >= 2 {
        let mut cs = s - 1;
        while cs > 0 {
            let a = r * (s - 1) - (s - 1 - cs);
            let b = r - 1;
            res.push((a, b));
            cs -= 1;
        }
        r -= 1;
    }

    println!("Case #{}: {}", i, res.len());
    for l in res {

    }
}
